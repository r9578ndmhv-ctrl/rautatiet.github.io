const API_URL = 'https://rata.digitraffic.fi/api/v1/metadata/stations'; const markers = []; let labelsVisible = true; let textScale = 1; let markerScale = 1; let markerColor = "#00B451"; function fitToVisible() { const visible = markers.filter(m => map.hasLayer(m.marker)); if (!visible.length) return; const group = L.featureGroup(visible.map(v => v.marker)); map.fitBounds(group.getBounds().pad(0.2)); } function showOnly(predicateFn) { markers.forEach(m => { const show = predicateFn(m.data); if (show) { if (!map.hasLayer(m.marker)) m.marker.addTo(map); } else { if (map.hasLayer(m.marker)) map.removeLayer(m.marker); } }); } function updateTooltips() { markers.forEach(m => { if (labelsVisible && map.hasLayer(m.marker)) { m.marker.openTooltip(); } else { m.marker.closeTooltip(); } }); } function updateSizes() { const zoom = map.getZoom(); const radius = Math.max(4, zoom * 1.5 * markerScale); const fontSize = Math.max(12, zoom * 1.6 * textScale); markers.forEach(m => { m.marker.setRadius(radius); m.marker.setStyle({ color: markerColor, fillColor: markerColor }); const el = m.marker.getTooltip()?.getElement(); if (el) el.style.fontSize = fontSize + "px"; }); } async function loadStations() { const res = await fetch(API_URL); const dataRaw = await res.json(); const data = dataRaw.filter(st => st.countryCode === "FI"); data.sort((a,b)=>a.stationName.localeCompare(b.stationName)); data.forEach(st => { if (!st.latitude || !st.longitude) return; const marker = L.circleMarker([st.latitude, st.longitude], { radius: 8, color: markerColor, fillColor: markerColor, fillOpacity: 0.9, weight: 1 }); marker.bindPopup(`${st.stationName}
Koodi: ${st.stationShortCode}`); marker.bindTooltip( `${st.stationName} (${st.stationShortCode})`, { permanent:true, direction:'top', offset:[0,-6], className:'station-label' } ); marker.addTo(map); markers.push({ data:st, marker }); const option = new Option( `${st.stationName} (${st.stationShortCode})`, st.stationShortCode, false,false ); $('#stationSelect').append(option); }); fitToVisible(); updateTooltips(); updateSizes(); } $('#stationSelect').select2({ placeholder:"Valitse asema(t)...", allowClear:true }); $('#stationSelect').on('change', function(){ const selected = $(this).val(); if (!selected || !selected.length) { showOnly(()=>true); } else { showOnly(st=>selected.includes(st.stationShortCode)); } fitToVisible(); updateTooltips(); }); document.getElementById('clear').addEventListener('click', ()=>{ $('#stationSelect').val(null).trigger('change'); showOnly(()=>true); fitToVisible(); updateTooltips(); }); document.getElementById('toggleLabels').addEventListener('change', e=>{ labelsVisible = e.target.checked; updateTooltips(); }); /* 🔘 Tekstin koko */ document.getElementById('textPlus').onclick = ()=>{ textScale += 0.2; updateSizes(); }; document.getElementById('textMinus').onclick = ()=>{ textScale = Math.max(0.4, textScale - 0.2); updateSizes(); }; /* 🔵 Pallon koko */ document.getElementById('markerPlus').onclick = ()=>{ markerScale += 0.2; updateSizes(); }; document.getElementById('markerMinus').onclick = ()=>{ markerScale = Math.max(0.4, markerScale - 0.2); updateSizes(); }; /* 🎨 Väri */ document.getElementById('colorPicker').addEventListener('input', e=>{ markerColor = e.target.value; updateSizes(); }); map.on('zoomend', updateSizes); loadStations();
